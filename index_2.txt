  <script>
    let port = null;
    let writer = null;
    let reader = null;
    let readLoopRunning = false;

    let alertBusy = false;
    let alertToken = 0;

    const statusEl = document.getElementById("status");
    const logEl = document.getElementById("log");
    const alertBox = document.getElementById("alertBox");
    const alarmLabel = document.getElementById("alarmLabel");
    const sectorCard = document.getElementById("sectorCard");
    const sectorGrid = document.getElementById("sectorGrid");

    function log(msg) {
      const time = new Date().toLocaleTimeString();
      logEl.textContent += `[${time}] ${msg}\n`;
      logEl.scrollTop = logEl.scrollHeight;
    }

    function clearLog() {
      logEl.textContent = "";
      log("Log limpiado");
    }

    function setStatus(msg) {
      statusEl.textContent = msg;
    }

    function setAlarmLabel(text) {
      alarmLabel.textContent = text;
    }

    function sleep(ms) {
      return new Promise(resolve => setTimeout(resolve, ms));
    }

    function setBlinkState(on) {
      alertBox.classList.toggle("on", on);
    }

    function clearBlink() {
      setBlinkState(false);
    }

    function resetAlarmVisual() {
      clearBlink();
      setAlarmLabel("SIN EVENTO");
    }

    async function blink(onMs, offMs, times, tokenRef) {
      for (let i = 0; i < times; i++) {
        if (tokenRef !== alertToken) return;
        setBlinkState(true);
        await sleep(onMs);

        if (tokenRef !== alertToken) return;
        setBlinkState(false);
        await sleep(offMs);
      }
    }

    async function runFirePattern(sector, tokenRef) {
      setAlarmLabel(`INCENDIO - SECTOR ${sector}`);

      for (let rep = 0; rep < 2; rep++) {
        if (tokenRef !== alertToken) return;

        await blink(1600, 1600, 3, tokenRef);

        if (tokenRef !== alertToken) return;

        setBlinkState(false);
        await sleep(2500);

        if (tokenRef !== alertToken) return;

        await blink(700, 700, sector, tokenRef);
      }

      if (tokenRef === alertToken) resetAlarmVisual();
    }

    async function runBlastPattern(tokenRef) {
      setAlarmLabel("QUEMADA");

      for (let rep = 0; rep < 2; rep++) {
        const start = Date.now();

        while ((Date.now() - start) < 10000) {
          if (tokenRef !== alertToken) return;

          setBlinkState(true);
          await sleep(1000);

          if (tokenRef !== alertToken) return;

          setBlinkState(false);
          await sleep(1000);
        }
      }

      if (tokenRef === alertToken) resetAlarmVisual();
    }

    async function runHomaqPattern(tokenRef) {
      setAlarmLabel("HOMAQ");

      for (let rep = 0; rep < 2; rep++) {
        const start = Date.now();

        while ((Date.now() - start) < 10000) {
          if (tokenRef !== alertToken) return;

          setBlinkState(true);
          await sleep(500);

          if (tokenRef !== alertToken) return;

          setBlinkState(false);
          await sleep(500);
        }
      }

      if (tokenRef === alertToken) resetAlarmVisual();
    }

    async function ejecutarAlertaVisual(major, minor) {
      alertToken++;
      const myToken = alertToken;
      alertBusy = true;

      try {
        resetAlarmVisual();

        if (major === 100 && minor >= 1 && minor <= 9) {
          log(`Blink local: INCENDIO sector ${minor}`);
          await runFirePattern(minor, myToken);
        } else if (major === 200 && minor === 10) {
          log("Blink local: QUEMADA");
          await runBlastPattern(myToken);
        } else if (major === 300 && minor === 20) {
          log("Blink local: HOMAQ");
          await runHomaqPattern(myToken);
        } else {
          setAlarmLabel(`SIN PATRÓN ${major},${minor}`);
          log(`Sin patrón visual para ${major},${minor}`);
        }
      } finally {
        if (myToken === alertToken) {
          alertBusy = false;
          resetAlarmVisual();
        }
      }
    }

    function cancelarAlertaVisual() {
      alertToken++;
      alertBusy = false;
      resetAlarmVisual();
      log("Blink local detenido");
    }

    async function connectSerial() {
      try {
        port = await navigator.serial.requestPort();
        await port.open({ baudRate: 115200 });

        const encoder = new TextEncoderStream();
        encoder.readable.pipeTo(port.writable);
        writer = encoder.writable.getWriter();

        const decoder = new TextDecoderStream();
        port.readable.pipeTo(decoder.writable);
        reader = decoder.readable.getReader();

        setStatus("Conectado a 115200");
        log("Puerto serial conectado");
        readSerialLoop();
      } catch (err) {
        log("Error conectando serial: " + err.message);
      }
    }

    async function sendSerialLine(line) {
      if (!writer) {
        log("Serial no conectado");
        return false;
      }

      try {
        await writer.write(line + "\n");
        log("TX -> " + line);
        return true;
      } catch (err) {
        log("Error TX: " + err.message);
        return false;
      }
    }

    async function readSerialLoop() {
      if (!reader || readLoopRunning) return;
      readLoopRunning = true;

      let buffer = "";

      try {
        while (true) {
          const { value, done } = await reader.read();
          if (done) break;
          if (!value) continue;

          buffer += value;

          let idx;
          while ((idx = buffer.indexOf("\n")) >= 0) {
            const line = buffer.slice(0, idx).replace(/\r/g, "").trim();
            buffer = buffer.slice(idx + 1);

            if (line) log("RX <- " + line);
          }
        }
      } catch (err) {
        log("Error RX: " + err.message);
      } finally {
        readLoopRunning = false;
      }
    }

    async function lanzarComando(major, minor) {
      const cmd = `${major},${minor}`;
      const ok = await sendSerialLine(cmd);
      if (!ok) return;

      await ejecutarAlertaVisual(major, minor);
    }

    document.getElementById("btnConnect").addEventListener("click", async () => {
      if (!("serial" in navigator)) {
        alert("Tu navegador no soporta Web Serial. Usa Chrome o Edge.");
        return;
      }
      await connectSerial();
    });

    document.getElementById("btnFire").addEventListener("click", () => {
      sectorCard.classList.remove("hidden");
      log("Selecciona sector para INCENDIO");
    });

    document.getElementById("btnBurn").addEventListener("click", async () => {
      sectorCard.classList.add("hidden");
      await lanzarComando(200, 10);
    });

    document.getElementById("btnHomaq").addEventListener("click", async () => {
      sectorCard.classList.add("hidden");
      await lanzarComando(300, 20);
    });

    document.getElementById("btnStop").addEventListener("click", async () => {
      sectorCard.classList.add("hidden");
      await sendSerialLine("stop");
      cancelarAlertaVisual();
    });

    document.getElementById("btnClear").addEventListener("click", () => {
      clearLog();
    });

    for (let i = 1; i <= 9; i++) {
      const btn = document.createElement("div");
      btn.className = "sector-btn";
      btn.textContent = `Sector ${i}`;
      btn.addEventListener("click", async () => {
        sectorCard.classList.add("hidden");
        await lanzarComando(100, i);
      });
      sectorGrid.appendChild(btn);
    }

    resetAlarmVisual();
  </script>
